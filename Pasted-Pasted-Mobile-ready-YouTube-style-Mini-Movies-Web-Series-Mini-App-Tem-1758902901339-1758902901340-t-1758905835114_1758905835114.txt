let adPlaying=false;
let midRollsTimes=[600,900]; // 10 min, 15 min
let skipTime=25;
let video=document.getElementById("myVideo");
let adContainer=document.getElementById("ad-overlay");

video.addEventListener("timeupdate",()=>{
    if(adPlaying) return;
    midRollsTimes.forEach((time,index)=>{
        if(video.currentTime>=time){
            adPlaying=true;
            video.pause();
            showMidRollAd(()=>{
                adPlaying=false;
                video.play();
                midRollsTimes.splice(index,1);
            });
        }
    });
});

function showPreRollAd(callback){
    adPlaying=true;
    adContainer.style.display="flex";
    adContainer.innerHTML="<p>Loading Pre-roll Ad...</p>";
    setTimeout(()=>{
        adContainer.style.display="none";
        adPlaying=false;
        callback();
    },5000);
}

function showMidRollAd(callback){
    adContainer.style.display="flex";

    adContainer.appendChild(skipBtn);
    let countdown=skipTime;
    let interval=setInterval(()=>{
        countdown--;
        skipBtn.innerText=countdown>0?`Skip Ad in ${countdown} sec`:"Skip Ad";
        if(countdown<=0) skipBtn.disabled=false;
    },1000);
    skipBtn.onclick=()=>{
        clearInterval(interval);
        adContainer.style.display="none";
        callback();
    };
    setTimeout(()=>{
        clearInterval(interval);
        adContainer.style.display="none";
        callback();
    },30000);
}
